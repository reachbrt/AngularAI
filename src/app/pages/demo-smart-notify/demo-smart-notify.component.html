Code Example

import { SmartNotifyService, NotificationContainerComponent } from '@angularai/smart-notify';

@Component({
  imports: [NotificationContainerComponent],
  template: `<ai-notification-container></ai-notification-container>`
})
export class MyComponent {
  constructor(private notify: SmartNotifyService) {}

  showNotification() {
    this.notify.success('Done!', 'Task completed.');
    this.notify.ai('Insight', 'AI suggestion here.');
  }
}
