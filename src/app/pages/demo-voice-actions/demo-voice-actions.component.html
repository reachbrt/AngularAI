Code Example

import { VoiceButtonComponent, VoiceActionsService } from '@angularai/voice-actions';

@Component({
  imports: [VoiceButtonComponent],
  template: `
    <ai-voice-button
      (transcript)="onTranscript($event)"
      (command)="onCommand($event)">
    </ai-voice-button>
  `
})
export class MyComponent {
  constructor(private voice: VoiceActionsService) {
    voice.registerCommand({ phrase: 'search', action: 'open_search' });
  }
}
